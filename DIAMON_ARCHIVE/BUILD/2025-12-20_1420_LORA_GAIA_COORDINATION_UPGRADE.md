// Sends feedback to Orange Pi 5 via LoRa

#include <LoRa.h>
#include <Wire.h>
#include <HMC5883L.h>  // Magnetometer

#define LORA_SS 5
#define LORA_RST 14
#define LORA_DIO0 2

HMC5883L magnetometer;
float baseline_field = 0.0;  // µT

void setup() {
  // LoRa init
  LoRa.setPins(LORA_SS, LORA_RST, LORA_DIO0);
  LoRa.begin(868E6);  // 868 MHz EU
  LoRa.setSpreadingFactor(7);  // Fast, short range adequate
  LoRa.setSignalBandwidth(125E3);

  // Magnetometer init
  Wire.begin();
  magnetometer.initialize();

  // Measure baseline (10 sec average)
  for (int i = 0; i < 100; i++) {
    Vector mag = magnetometer.readRaw();
    baseline_field += sqrt(mag.XAxis*mag.XAxis + mag.YAxis*mag.YAxis + mag.ZAxis*mag.ZAxis);

}
